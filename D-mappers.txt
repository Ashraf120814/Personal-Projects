        // Particle network animation for hero (advanced visual)
        function createParticleNetwork() {
            const canvas = document.getElementById('particle-canvas')
            if (!canvas) return
            const ctx = canvas.getContext('2d')
            
            let width = canvas.offsetWidth
            let height = canvas.offsetHeight
            
            canvas.width = width
            canvas.height = height
            
            // Particle class
            class Particle {
                constructor() {
                    this.x = Math.random() * width
                    this.y = Math.random() * height
                    this.vx = (Math.random() - 0.5) * 0.8
                    this.vy = (Math.random() - 0.5) * 0.8
                    this.radius = Math.random() * 3.5 + 1
                }
                
                update() {
                    this.x += this.vx
                    this.y += this.vy
                    
                    // Bounce off edges
                    if (this.x < 0 || this.x > width) this.vx *= -1
                    if (this.y < 0 || this.y > height) this.vy *= -1
                }
            }
            
            const particles = []
            const particleCount = 110
            
            for (let i = 0; i < particleCount; i++) {
                particles.push(new Particle())
            }
            
            function connectParticles() {
                for (let i = 0; i < particles.length; i++) {
                    for (let j = i + 1; j < particles.length; j++) {
                        const dx = particles[i].x - particles[j].x
                        const dy = particles[i].y - particles[j].y
                        const distance = Math.sqrt(dx * dx + dy * dy)
                        
                        if (distance < 130) {
                            ctx.strokeStyle = `rgba(0, 245, 212, ${1 - distance / 130})`
                            ctx.lineWidth = 0.8
                            ctx.beginPath()
                            ctx.moveTo(particles[i].x, particles[i].y)
                            ctx.lineTo(particles[j].x, particles[j].y)
                            ctx.stroke()
                        }
                    }
                }
            }
            
            function animateParticles() {
                ctx.clearRect(0, 0, width, height)
                
                // Draw particles
                for (let i = 0; i < particles.length; i++) {
                    const p = particles[i]
                    p.update()
                    
                    ctx.fillStyle = '#00f5d4'
                    ctx.shadowBlur = 12
                    ctx.shadowColor = '#00f5d4'
                    ctx.beginPath()
                    ctx.arc(p.x, p.y, p.radius, 0, Math.PI * 2)
                    ctx.fill()
                }
                
                connectParticles()
                requestAnimationFrame(animateParticles)
            }
            
            // Handle resize
            window.addEventListener('resize', () => {
                width = canvas.offsetWidth
                height = canvas.offsetHeight
                canvas.width = width
                canvas.height = height
            })
            
            animateParticles()
            
            // Console signature
            console.log('%c✅ D-Mappers website live • Built with premium animations by Stark', 'background:#00f5d4;color:#020617;font-size:10px;padding:1px 3px;border-radius:2px')
        }
        
        // Mini interactive demo canvas
        function createDemoCanvas() {
            const canvas = document.getElementById('demo-canvas')
            if (!canvas) return
            const ctx = canvas.getContext('2d')
            
            let nodes = []
            const colors = ['#00f5d4', '#10b981', '#a78bfa']
            
            class Node {
                constructor(x, y) {
                    this.x = x
                    this.y = y
                    this.vx = (Math.random() - 0.5) * 1.5
                    this.vy = (Math.random() - 0.5) * 1.5
                    this.color = colors[Math.floor(Math.random() * colors.length)]
                    this.radius = 9
                }
            }
            
            // Initial nodes
            function initNodes() {
                nodes = []
                for (let i = 0; i < 18; i++) {
                    nodes.push(new Node(
                        Math.random() * canvas.width,
                        Math.random() * canvas.height
                    ))
                }
            }
            
            function draw() {
                ctx.clearRect(0, 0, canvas.width, canvas.height)
                
                // Draw connections
                ctx.lineWidth = 1.8
                for (let i = 0; i < nodes.length; i++) {
                    for (let j = i + 1; j < nodes.length; j++) {
                        const dx = nodes[i].x - nodes[j].x
                        const dy = nodes[i].y - nodes[j].y
                        const dist = Math.hypot(dx, dy)
                        if (dist < 160) {
                            ctx.strokeStyle = `rgba(16, 185, 129, ${0.6 - dist / 180})`
                            ctx.beginPath()
                            ctx.moveTo(nodes[i].x, nodes[i].y)
                            ctx.lineTo(nodes[j].x, nodes[j].y)
                            ctx.stroke()
                        }
                    }
                }
                
                // Draw & update nodes
                nodes.forEach(node => {
                    node.x += node.vx
                    node.y += node.vy
                    
                    // Bounce
                    if (node.x < node.radius || node.x > canvas.width - node.radius) node.vx *= -1
                    if (node.y < node.radius || node.y > canvas.height - node.radius) node.vy *= -1
                    
                    // Draw node
                    ctx.shadowBlur = 25
                    ctx.shadowColor = node.color
                    ctx.fillStyle = node.color
                    ctx.beginPath()
                    ctx.arc(node.x, node.y, node.radius, 0, Math.PI * 2)
                    ctx.fill()
                    
                    // Inner dot
                    ctx.shadowBlur = 0
                    ctx.fillStyle = '#fff'
                    ctx.beginPath()
                    ctx.arc(node.x, node.y, 3, 0, Math.PI * 2)
                    ctx.fill()
                })
                
                requestAnimationFrame(draw)
            }
            
            // Mouse interaction
            canvas.addEventListener('mousemove', (e) => {
                const rect = canvas.getBoundingClientRect()
                const mx = e.clientX - rect.left
                const my = e.clientY - rect.top
                
                // Push nodes away from mouse
                nodes.forEach(node => {
                    const dx = node.x - mx
                    const dy = node.y - my
                    const dist = Math.hypot(dx, dy)
                    if (dist < 110) {
                        node.vx += dx / dist * 3
                        node.vy += dy / dist * 3
                    }
                })
            })
            
            // Initialize and run
            initNodes()
            draw()
            
            // Occasional random node addition
            setInterval(() => {
                if (nodes.length < 25) {
                    nodes.push(new Node(Math.random() * canvas.width, Math.random() * canvas.height))
                }
            }, 2100)
        }
        
        // Animate stats on scroll
        function animateStats() {
            const stats = [
                { id: 'stat-1', target: 248000, suffix: '', duration: 2200 },
                { id: 'stat-2', target: 142, suffix: '', duration: 1600 },
                { id: 'stat-3', target: 99.97, suffix: '%', duration: 1400 },
                { id: 'stat-4', target: 4.2, suffix: 's', duration: 1800 },
                { id: 'stat-5', target: 87, suffix: '', duration: 1300 },
                { id: 'stat-6', target: 98000000, suffix: '', prefix: '₹', duration: 2400 }
            ]
            
            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if (entry.isIntersecting) {
                        stats.forEach(stat => {
                            const el = document.getElementById(stat.id)
                            if (!el) return
                            let start = 0
                            const increment = stat.target / (stat.duration / 16)
                            
                            const timer = setInterval(() => {
                                start += increment
                                if (start >= stat.target) {
                                    clearInterval(timer)
                                    start = stat.target
                                }
                                let display = Math.floor(start)
                                if (stat.id === 'stat-3') display = start.toFixed(2)
                                if (stat.id === 'stat-4') display = start.toFixed(1)
                                el.textContent = (stat.prefix || '') + display + (stat.suffix || '')
                            }, 16)
                        })
                        observer.disconnect()
                    }
                })
            }, { threshold: 0.8 })
            
            const statsContainer = document.querySelector('.grid-cols-3')
            if (statsContainer) observer.observe(statsContainer)
        }
